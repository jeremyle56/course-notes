\section{Products of Groups}

Recall given groups \(G_1, \dots, G_n\), the set \(G_1 \times G_2 \times \dots G_n = \{(g_1, \dots, g_n): g_1 \in G_1, \dots g_n \in G_n \}\). More generally if \(G_i, i \in I\) are groups then \(\prod_{i\in I} G_i = \{(g_i)_{i \in I}: g_i \in G_i\}\).

\begin{prop-defn}[Product]
The set \(\prod_{i \in I}  G_i\) is called the (direct) product of the \(G_i\)'s, it is a group when endowed with co-ordinatewise multiplication. \((g_i)(g'_i) = (g_ig_i')\)
\begin{enumerate}
    \item \(1_G = (1_{G_i}) = (1_{G_1}, 1_{G_2}, 1_{G_3}, \dots)\)
    \item \((g_i)^{-1} = (g_i^{-1})\)
\end{enumerate}
\end{prop-defn}

\exmp{
    Consider \(\bZ^2 = \bZ \times \bZ\). \((a, b) + (a', b') = (a + a', b + b')\), group law in each coordinate. \(\bZ^2 = \langle (1, 0), (0, 1) \rangle\) is finitely generated.
}

\begin{proposition}[Canonical Injections and Projections]
    Let \(G_i, i \in I\) be groups and \(r \in I\).
    \begin{enumerate}
        \item The canonical injection \(\iota_r : G_n \to \prod_{i \in I}G_i; g \mapsto (g_i)_{i \in I}\) where \(g_i = 1\) if \(i \neq r\) or \(g_i = g\) if \(i = r\).
        \item The canonical project \(\pi_r: \prod_{i \in I} G_i \to G_r; (g_i)_{i \in I} \mapsto g_r\).
        \item \(\frac{G_1 \times G_2}{G_1 \times \{1\}} \cong G_2\) (Note: \(G_n \times \{1\} \normal G_1 \times G_2\)).
              \proof{
                  \(\pi_2: G_1 \times G_2 \to G_2\). Apply First Isomorphism Theorem
              }
    \end{enumerate}
\end{proposition}

\begin{proposition}[Internal Characterisation of Product]
    Let \(G_1, \dots, G_n \leq G\). Assume \(G = \langle G_1, \dots, G_n \rangle\). Assume:
    \begin{enumerate}
        \item If \(i \neq j\) then elements of \(G_i\) and \(G_j\) commute
        \item For any \(i\), \(G_i \cap \langle U_{\ell \neq i} G_\ell \rangle = 1\).
    \end{enumerate}
    Then there is an isomorphism \(\phi: G_1 \times \dots G_n \to G; (g_1, \dots, g_n) \mapsto g_1g_2 \cdots g_n\).
    \proof{
    Check homomorphism:
    \begin{align*}
        \phi((g_1, \dots, g_n)(h_1, \dots, h_n)) & = \phi((g_1h_1, \dots g_nh_n))                  \\
                                                 & = g_1h_1g_2h_2 \cdots g_nh_n                    \\
                                                 & = g_1 \cdots g_n h_1 \cdots h_n \tag{using (i)} \\
                                                 & = \phi(g_1 \dots g_n)\phi(h_1 \dots h_n)
    \end{align*}
    Surjective? Yes because \(G\) is generated by \(G_1, \dots, G_n\). Injective? Suppose \(\phi((g_1, \dots, g_n)) = 1\), then \(g_1 \cdots g_n = 1 \implies g_1^{-1} \in G_1 = g_2 \cdots g_n \in \langle G_2 \cdots G_n \rangle\) by (ii) must be \(\id\). So \(g_1 = 1\) and \(g_2 \cdots g_n = 1\). Repeat the same argument to get all \(g_i = 1\).
    }
\end{proposition}

\begin{corollary}
    Let \(G\) = finite group of exponent 2. i.e. LCM of all orders of group element is 2. Then \(G \cong \bZ / 2\bZ \times \cdots \bZ / 2\bZ\).
    \proof{
    \(G\) is finitely genereqated. Choose minimal generating set \(\{g_1, \dots, g_n\}\), each \(\langle g_i \rangle \cong \bZ / 2\bZ\). Want to prove that \(G \cong \langle g_1 \rangle \times \dots \langle g_n \rangle\).  Condition (i): Need \(g_i g_j = g_j g_i\) for \(i \neq j\). \(\ord(g_i g_j) = 2\), so \(g_i g_j g_i g_j = 1 \implies g_i g_j = g_j^{-1}g_i^{-1} = g_j g_i\). Condition (ii): e.g. \(\langle g_1 \rangle \cap \langle g_2, \dots , g_n \rangle = \{1\}\). If false, then \(g_1 \in \langle g_2, \dots, g_n \rangle\) but then our generating set is not minimal. By proposition \(G \cong \langle g_1 \rangle \times \cdots \times \langle g_n \rangle\).
    }
\end{corollary}

\begin{theorem}
    Let \(G\) be a finitely generated abelian group. Then \(G \cong\) product of cyclic groups. In fact \(G \cong \bZ / h_1 \bZ \times \bZ / g_2 \bZ \times \cdots \times \bZ / h_n \bZ \times \bZ^s\) where \(h_1 \mid h_2 \mid h_3 \mid \dots \mid h_n\) for some \(n, r \in \bN\).
\end{theorem}
